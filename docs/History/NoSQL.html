def self.fetch(user, page = 0, per_page = 9)
	results = []
	offset = page * per_page
	per_page = (offset + per_page) if page > 0

	results = redis.lrange key(user), offset, per_page

	if results.present?
		results = build_result_list(user, results)
	end

	results
end
